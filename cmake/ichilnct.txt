/*
 * Stub for the four AuxInfo<->Input InChI API entry points.
 *
 * Open Babel 3.2.0 ships a vendored InChI 1.07.5 that dropped ichilnct.c
 * but kept the cdecl_/pasc_ wrappers in inchi_dll.c referencing
 *   Get_inchi_Input_FromAuxInfo
 *   Get_std_inchi_Input_FromAuxInfo
 *   Free_inchi_Input
 *   Free_std_inchi_Input
 * MSVC refuses to link the inchi DLL with these unresolved externals;
 * GCC/Clang accept them as lazily resolved at runtime. Until openbabel
 * either restores the real implementation (with inchi_dll_b.c) or removes
 * the wrappers from inchi_dll.c, this stub satisfies the DLL exports.
 *
 * Nothing inside openbabel itself calls these APIs; they are reachable
 * only via the public DLL, where AuxInfo->Input round-tripping is not
 * supported in this build.
 */

#include "mode.h"
#include "inchi_api.h"

EXPIMP_TEMPLATE INCHI_API int INCHI_DECL Get_std_inchi_Input_FromAuxInfo(
    char *szInchiAuxInfo,
    int bDoNotAddH,
    InchiInpData *pInchiInp )
{
    (void) szInchiAuxInfo;
    (void) bDoNotAddH;
    (void) pInchiInp;
    return inchi_Ret_ERROR;
}

EXPIMP_TEMPLATE INCHI_API int INCHI_DECL Get_inchi_Input_FromAuxInfo(
    char *szInchiAuxInfo,
    int bDoNotAddH,
    int bDiffUnkUndfStereo,
    InchiInpData *pInchiInp )
{
    (void) szInchiAuxInfo;
    (void) bDoNotAddH;
    (void) bDiffUnkUndfStereo;
    (void) pInchiInp;
    return inchi_Ret_ERROR;
}

void INCHI_DECL Free_std_inchi_Input( inchi_Input *pInp )
{
    (void) pInp;
}

void INCHI_DECL Free_inchi_Input( inchi_Input *pInp )
{
    (void) pInp;
}
